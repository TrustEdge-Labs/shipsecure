Create `frontend/app/actions/scan.ts` with a Server Action for form submission.

```typescript
'use server'

import { z } from 'zod'
import { redirect } from 'next/navigation'

const scanSchema = z.object({
  url: z.string()
    .min(1, 'URL is required')
    .url('Please enter a valid URL (e.g., https://example.com)')
    .refine(url => url.startsWith('http://') || url.startsWith('https://'), {
      message: 'URL must start with http:// or https://'
    }),
  email: z.string()
    .min(1, 'Email is required')
    .email('Please enter a valid email address'),
})

export interface ScanFormState {
  errors?: {
    url?: string[]
    email?: string[]
    _form?: string[]
  }
  scanId?: string
}

export async function submitScan(
  prevState: ScanFormState,
  formData: FormData
): Promise<ScanFormState> {
  // Validate input
  const validatedFields = scanSchema.safeParse({
    url: formData.get('url'),
    email: formData.get('email'),
  })

  if (!validatedFields.success) {
    return {
      errors: validatedFields.error.flatten().fieldErrors,
    }
  }

  // Call Rust backend
  const backendUrl = process.env.BACKEND_URL || 'http://localhost:3000'

  try {
    const response = await fetch(`${backendUrl}/api/v1/scans`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(validatedFields.data),
    })

    if (!response.ok) {
      const errorBody = await response.json().catch(() => ({}))
      const detail = errorBody.detail || 'Failed to start scan. Please try again.'

      // Handle rate limit specifically
      if (response.status === 429) {
        return {
          errors: { _form: ['You have reached the maximum number of scans for today. Please try again tomorrow.'] }
        }
      }

      return {
        errors: { _form: [detail] }
      }
    }

    const data = await response.json()

    // Return scanId - the client component will handle the redirect with a delay
    return { scanId: data.id }

  } catch (error) {
    return {
      errors: { _form: ['Unable to connect to the scanning service. Please try again later.'] }
    }
  }
}
```

Note: Use `BACKEND_URL` (not `NEXT_PUBLIC_BACKEND_URL`) in the Server Action because it runs on the server and does NOT need to be exposed to the client. This is the server-to-server URL (e.g., `http://localhost:3000` in dev).

The redirect happens client-side after showing the "Scan started!" message for 2-3 seconds (per CONTEXT.md decision). The Server Action returns the scanId and the client component handles the delayed redirect.

**Create `frontend/components/scan-form.tsx`** (client component):
```typescript
'use client'

import { useActionState, useEffect } from 'react'
import { useRouter } from 'next/navigation'
import { submitScan, type ScanFormState } from '@/app/actions/scan'

export function ScanForm() {
  const [state, formAction, pending] = useActionState(submitScan, {} as ScanFormState)
  const router = useRouter()

  // Handle successful submission with delayed redirect
  useEffect(() => {
    if (state.scanId) {
      const timer = setTimeout(() => {
        router.push(`/scan/${state.scanId}`)
      }, 2500) // 2.5 second delay per CONTEXT.md
      return () => clearTimeout(timer)
    }
  }, [state.scanId, router])

  // If scan started, show confirmation
  if (state.scanId) {
    return (
      <div className="text-center p-8 bg-green-50 dark:bg-green-950 rounded-xl border border-green-200 dark:border-green-800">
        <div className="text-4xl mb-3">&#10003;</div>
        <h2 className="text-xl font-semibold text-green-800 dark:text-green-200 mb-2">
          Scan started!
        </h2>
        <p className="text-green-600 dark:text-green-400">
          Redirecting to your scan progress...
        </p>
      </div>
    )
  }

  return (
    <form action={formAction} className="space-y-4">
      {/* Form-level errors */}
      {state.errors?._form && (
        <div className="p-3 bg-red-50 dark:bg-red-950 border border-red-200 dark:border-red-800 rounded-lg text-red-700 dark:text-red-300 text-sm">
          {state.errors._form[0]}
        </div>
      )}

      {/* URL input */}
      <div>
        <label htmlFor="url" className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1">
          Website URL
        </label>
        <input
          id="url"
          name="url"
          type="url"
          placeholder="https://your-app.vercel.app"
          required
          className="w-full px-4 py-3 rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-800 text-gray-900 dark:text-gray-100 focus:ring-2 focus:ring-blue-500 focus:border-blue-500 outline-none transition"
        />
        {state.errors?.url && (
          <p className="mt-1 text-sm text-red-600 dark:text-red-400">{state.errors.url[0]}</p>
        )}
      </div>

      {/* Email input */}
      <div>
        <label htmlFor="email" className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1">
          Email (for results notification)
        </label>
        <input
          id="email"
          name="email"
          type="email"
          placeholder="you@example.com"
          required
          className="w-full px-4 py-3 rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-800 text-gray-900 dark:text-gray-100 focus:ring-2 focus:ring-blue-500 focus:border-blue-500 outline-none transition"
        />
        {state.errors?.email && (
          <p className="mt-1 text-sm text-red-600 dark:text-red-400">{state.errors.email[0]}</p>
        )}
      </div>

      {/* Submit button */}
      <button
        type="submit"
        disabled={pending}
        className="w-full py-3 px-6 rounded-lg bg-blue-600 hover:bg-blue-700 disabled:bg-blue-400 text-white font-semibold transition text-lg"
      >
        {pending ? 'Starting scan...' : 'Scan Now — Free'}
      </button>
    </form>
  )
}
```

async function getScanCount(): Promise<number | null> {
  try {
    const backendUrl = process.env.BACKEND_URL || 'http://localhost:3000'
    const res = await fetch(`${backendUrl}/api/v1/stats/scan-count`, {
      next: { revalidate: 60 } // Cache for 60 seconds
    })
    if (!res.ok) return null
    const data = await res.json()
    return data.count
  } catch {
    return null
  }
}

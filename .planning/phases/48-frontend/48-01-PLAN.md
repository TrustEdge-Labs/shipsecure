// Match DepFinding from src/scanners/lockfile_parser.rs
export interface SupplyChainFinding {
  name: string
  version: string
  osv_id: string
  description: string
  tier: 'Infected' | 'Vulnerable' | 'Advisory'
}

// Match ParsedDep for unscanned items
export interface UnscannedDep {
  name: string
  version: string
  source: 'Git' | 'File' | 'Link' | 'Tarball'
  is_dev: boolean
}

// Match SupplyChainScanResult from backend
export interface SupplyChainResults {
  total_deps: number
  infected: SupplyChainFinding[]
  vulnerable: SupplyChainFinding[]
  advisory: SupplyChainFinding[]
  no_known_issues: string[]  // "name@version" strings
  unscanned: UnscannedDep[]
  scanned_at: string
}

// POST /api/v1/scans/supply-chain response
export interface SupplyChainScanResponse {
  status: string
  results_token: string | null
  share_url: string | null
  share_unavailable: boolean
  results: SupplyChainResults
}

// GET /api/v1/results/:token response for supply chain scans
export interface SupplyChainResultsPageData {
  id: string
  target_url: string
  status: string
  kind: string
  expires_at: string | null
  created_at: string
  completed_at: string | null
  supply_chain_results: SupplyChainResults | null
}

// Error response (RFC 7807)
export interface ApiErrorResponse {
  type: string
  title: string
  status: number
  detail: string
}

**2. Create `frontend/app/actions/supply-chain-scan.ts`** — Server action per D-02.

This is a client-side fetch (NOT a Next.js server action) because the form uses `useState` for tabs and needs to handle multipart uploads. Create an exported async function `submitSupplyChainScan` that:

- Accepts `mode: 'github' | 'upload' | 'paste'` and the input value (string for github/paste, File for upload)
- For `github` mode: POST JSON `{ "github_url": value }` with `Content-Type: application/json`
- For `paste` mode: POST JSON `{ "lockfile_content": value }` with `Content-Type: application/json`
- For `upload` mode: POST multipart/form-data with field name `lockfile`
- Target URL: `/api/v1/scans/supply-chain` (relative URL per project convention — Nginx proxies /api/ to backend)
- Parse success response as `SupplyChainScanResponse`
- Parse error responses: check `response.ok`, if false parse body as `ApiErrorResponse`, map to user-friendly messages per D-09:
  - Status 400 + type contains "invalid-lockfile": "This doesn't look like a valid package-lock.json file"
  - Status 400 + type contains "too-many-deps": "Too many dependencies (max 5,000)"
  - Status 502: "Something went wrong. Try again or upload your lockfile directly"
  - Status 504: "Scan timed out. Try a smaller lockfile"
  - Status 429: rate limit message from `detail` field
  - Other: "An unexpected error occurred"
- Return `{ data: SupplyChainScanResponse } | { error: string }`

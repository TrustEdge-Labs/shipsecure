Create `frontend/components/active-scans-poller.tsx` as a minimal client component whose sole job is to call `router.refresh()` on an interval.

Requirements:
- `'use client'` directive at top
- Import `useEffect` from `react` and `useRouter` from `next/navigation`
- Props: `{ hasActiveScans: boolean }` ‚Äî parent controls whether polling is active
- Logic: when `hasActiveScans` is true, set up a `setInterval` at 7000ms (7 seconds, within the 5-10s range) that calls `router.refresh()`; clear the interval on cleanup or when `hasActiveScans` becomes false
- Returns `null` ‚Äî this component renders nothing, it is a pure behavior island

```tsx
'use client'

import { useEffect } from 'react'
import { useRouter } from 'next/navigation'

interface ActiveScansPollertProps {
  hasActiveScans: boolean
}

export function ActiveScansPoller({ hasActiveScans }: ActiveScansPollertProps) {
  const router = useRouter()

  useEffect(() => {
    if (!hasActiveScans) return

    const interval = setInterval(() => {
      router.refresh()
    }, 7000)

    return () => clearInterval(interval)
  }, [hasActiveScans, router])

  return null
}
```

Note: `router.refresh()` in Next.js App Router re-fetches server components from the server using the current URL ‚Äî this re-runs the `DashboardPage` server component, fetching fresh data from the backend. When `active_scans` becomes empty (scan completes), the parent will pass `hasActiveScans={false}` and the interval will not be started on the next render.

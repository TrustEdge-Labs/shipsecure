```rust
// Spawn the paid scan + PDF + email pipeline
let pool_clone = pool.clone();
let orchestrator_clone = state.orchestrator.clone();
tokio::spawn(async move {
    // Step 1: Execute paid scan
    let scan = match db::scans::get_scan(&pool_clone, scan_id).await {
        Ok(Some(s)) => s,
        Ok(None) => { tracing::error!("Scan {} not found for paid audit", scan_id); return; }
        Err(e) => { tracing::error!("DB error for paid scan {}: {}", scan_id, e); return; }
    };

    // Run paid scan via orchestrator
    orchestrator_clone.spawn_paid_scan(scan_id, scan.target_url.clone());

    // Wait for scan completion (poll database)
    // Poll every 5 seconds for up to 15 minutes
    let max_wait = std::time::Duration::from_secs(900);
    let poll_interval = std::time::Duration::from_secs(5);
    let start = std::time::Instant::now();

    loop {
        tokio::time::sleep(poll_interval).await;
        if start.elapsed() > max_wait {
            tracing::error!("Paid scan {} timed out waiting for completion", scan_id);
            return;
        }

        match db::scans::get_scan(&pool_clone, scan_id).await {
            Ok(Some(s)) if s.status == ScanStatus::Completed || s.status == ScanStatus::Failed => {
                if s.status == ScanStatus::Failed {
                    tracing::error!("Paid scan {} failed: {:?}", scan_id, s.error_message);
                    // TODO: Send failure email, consider refund
                    return;
                }
                // Scan completed — proceed to PDF + email
                break;
            }
            _ => continue, // Still running
        }
    }

    // Step 2: Fetch findings for PDF
    let findings = match db::findings::get_findings_by_scan(&pool_clone, scan_id).await {
        Ok(f) => f,
        Err(e) => { tracing::error!("Failed to fetch findings for PDF: {}", e); return; }
    };

    let scan = db::scans::get_scan(&pool_clone, scan_id).await
        .ok().flatten().unwrap_or_else(|| { tracing::error!("Scan {} gone", scan_id); panic!() });

    // Step 3: Generate PDF
    let pdf_bytes = match crate::pdf::generate_report(
        &scan.target_url,
        scan.score.as_deref().unwrap_or("N/A"),
        &scan.completed_at.map(|d| d.format("%Y-%m-%d %H:%M UTC").to_string()).unwrap_or_default(),
        scan.detected_framework.as_deref(),
        scan.detected_platform.as_deref(),
        &findings,
    ) {
        Ok(bytes) => bytes,
        Err(e) => { tracing::error!("PDF generation failed for scan {}: {}", scan_id, e); return; }
    };

    // Mark PDF as generated
    let _ = db::paid_audits::mark_pdf_generated(&pool_clone, scan_id).await;

    // Step 4: Send email with PDF
    let summary = crate::email::FindingsSummary {
        critical: findings.iter().filter(|f| f.severity == crate::models::Severity::Critical).count() as i64,
        high: findings.iter().filter(|f| f.severity == crate::models::Severity::High).count() as i64,
        medium: findings.iter().filter(|f| f.severity == crate::models::Severity::Medium).count() as i64,
        low: findings.iter().filter(|f| f.severity == crate::models::Severity::Low).count() as i64,
        total: findings.len() as i64,
    };

    let base_url = std::env::var("TRUSTEDGE_BASE_URL").unwrap_or_else(|_| "http://localhost:3001".to_string());
    let results_token = scan.results_token.as_deref().unwrap_or("");

    if let Err(e) = crate::email::send_paid_audit_email(
        &customer_email,
        &scan.target_url,
        scan.score.as_deref().unwrap_or("N/A"),
        &summary,
        results_token,
        &base_url,
        pdf_bytes,
        &scan_id.to_string(),
    ).await {
        tracing::error!("Failed to send paid audit email for scan {}: {}", scan_id, e);
    }

    tracing::info!("Paid audit pipeline complete for scan {}", scan_id);
});
```

This is pseudo-code — adapt to the actual code structure. The key flow is: wait for scan completion -> generate PDF -> send email. The webhook returns 200 immediately; all this runs in the background.

NOTE: The `spawn_paid_scan` starts a NEW scan execution with paid tier. The orchestrator handles the scanning. The webhook pipeline above just waits for it to finish, then generates PDF and emails.

IMPORTANT: The paid scan creates NEW findings by re-running with extended parameters. However, the scan_id is the SAME as the original free scan. The orchestrator needs to handle this — either clear old findings first, or append new ones with deduplication. The simplest approach: Before starting the paid scan, have the pipeline clear the old findings (DELETE FROM findings WHERE scan_id = $1), reset scan status to pending, and re-run. This way the paid results page shows ALL findings (free + paid templates combined in one run).

export function mockSupplyChainScan(
  next: NextFixture,
  fixtures: SupplyChainFixtures
): void {
  next.onFetch(async (request) => {
    const url = new URL(request.url);
    if (url.pathname === '/api/v1/scans/supply-chain' && request.method === 'POST') {
      await new Promise((r) => setTimeout(r, 100));
      return new Response(JSON.stringify(fixtures.scanResponse), {
        status: 200,
        headers: { 'Content-Type': 'application/json' },
      });
    }
    return undefined;
  });
}

```typescript
import { test, expect } from 'next/experimental/testmode/playwright';
import { supplyChainFixtures } from './fixtures/supply-chain';
import { mockSupplyChainScan, mockSupplyChainResults } from './helpers/fetch-mocks';

test.describe('Supply Chain Scan', () => {

  test('happy path: paste lockfile content and see results', async ({ page, next }) => {
    // 1. Mock the scan endpoint and results endpoint
    mockSupplyChainScan(next, supplyChainFixtures);
    mockSupplyChainResults(next, supplyChainFixtures);

    // 2. Navigate to supply chain page
    await page.goto('/supply-chain');

    // 3. Verify page loaded
    await expect(page.locator('h1')).toBeVisible();

    // 4. Click 'Paste Content' tab
    await page.click('text=Paste Content');

    // 5. Fill in the paste textarea with a minimal lockfile
    const minimalLockfile = JSON.stringify({
      lockfileVersion: 3,
      packages: {
        "": { name: "test-app", version: "1.0.0" },
        "node_modules/lodash": { version: "4.17.20", resolved: "https://registry.npmjs.org/lodash/-/lodash-4.17.20.tgz" }
      }
    });
    await page.fill('#paste-content', minimalLockfile);

    // 6. Submit the form
    await page.click('text=Scan Dependencies');

    // 7. Wait for navigation to results page
    await page.waitForURL('**/supply-chain/results/**', { timeout: 15000 });

    // 8. Verify results page shows summary cards
    await expect(page.locator('text=Vulnerable')).toBeVisible();
    await expect(page.locator('text=No Known Issues')).toBeVisible();

    // 9. Verify the specific finding appears
    await expect(page.locator('text=lodash')).toBeVisible();
    await expect(page.locator('text=GHSA-jf85-cpcp-j695')).toBeVisible();
  });

  test('error state: empty paste content shows validation error', async ({ page, next }) => {
    // 1. Navigate to supply chain page
    await page.goto('/supply-chain');

    // 2. Click 'Paste Content' tab
    await page.click('text=Paste Content');

    // 3. Submit without entering any content
    await page.click('text=Scan Dependencies');

    // 4. Verify validation error appears
    await expect(page.locator('text=Please paste your package-lock.json content')).toBeVisible();
  });

});
```

The test uses the same `next/experimental/testmode/playwright` pattern as the existing `free-scan.spec.ts`. The happy path mocks both the scan submission and results page fetch. The error state test requires no mocking since validation happens client-side.

IMPORTANT: The `page.fill('#paste-content', ...)` targets the textarea by its `id="paste-content"` attribute which is set on the paste textarea in `supply-chain-form.tsx`.
